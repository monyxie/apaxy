	# ENABLE INDEXING
	Options +Indexes

	# INDEX OPTIONS
	IndexOptions +Charset=UTF-8 +FancyIndexing +IgnoreCase +FoldersFirst +XHTML +HTMLTable +SuppressRules +SuppressDescription +NameWidth=* +IconsAreLinks +AddAltClass

    	# META VIEWPORT
    	IndexHeadInsert "<meta name=\"viewport\" content=\"width=device-width, initial-scale=1\" />"

	# HIDE /theme DIRECTORY
	IndexIgnore .htaccess /theme

#
# GENERAL ICONS (BLANK, DIRECTORY, PARENT DIRECTORY)
#


    AddAlt ' dir' ^^DIRECTORY^^
	AddAlt ' parent' ..
	
	AddAltByType ' text'  text/*
	AddAltByType ' image' image/*
	AddAltByType ' audio' audio/*
	AddAltByType ' video' video/*

	HeaderName /theme/header.html
	ReadmeName /theme/footer.html
	IndexStyleSheet /theme/style.css
